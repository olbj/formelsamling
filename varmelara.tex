\section*{Grundläggande formler och definitioner}
\begin{flalign*}
	&\textbf{Första huvudsatsen för ett slutet system:}  & \\
	& Q - W = \Delta U & \\
	& \text{Med teckenkonventionen:} \\
	& Q > 0 \text{ betyder \textit{tillfört värme} från omgivningen} \\
	& Q < 0 \text{ betyder \textit{bortfört värme} från systemet} \\
	& W > 0 \text{ betyder \textit{bortfört arbete} från systemet} \\
	& W < 0 \text{ betyder \textit{tillfört arbete} från omgivningen} 
\end{flalign*}
\begin{flalign*}
	&\textbf{Första huvudsatsen för ett öppet system:}  & \\
	& q - w = (h_2 - h_1)_{entalpi} + \dfrac{v_2^2 - v_1^2}{2}+ g \cdot (h_2 - h_1)_{\text{höjd}}
\end{flalign*}
%\textbf{Första huvudsatsen för ett slutet system:} $ Q - W = \Delta U} $ \par 
	\textbf{Allmänna gaslagen: } $ p\cdot V = m \cdot R \cdot T $ , eller  $ p\cdot V = n \cdot R_u \cdot T $\\
	där $ R_u = \text{Almänna gaskonstanten och } n= \text{antal mol} $ \par
	\textbf{Värme vid konstant tryck: } \\
  $ dQ = m\cdot c_p\cdot dt$. $ Q \approx m \cdot \overline {c_p} \cdot \Delta T \text { då } c_p \text{ varierar lite.}$ \par
	\textbf{Värme vid konstant volymitet: } \\
  $ dQ = m\cdot c_v\cdot dt$. $ Q \approx m \cdot \overline {c_v} \cdot \Delta T \text { då } c_v \text{ varierar lite.}$ 
	\begin{flalign*}
    &\textbf{Samband mellan c\textsubscript{p}, c\textsubscript{v} och R för ideala gaser:}&\\
    &c_p - c_v = R \text{ samt } \dfrac{c_p}{c_v} = \kappa.&
	\end{flalign*}
	\textbf{Volymändringsarbete: } $; dW = F \cdot ds = p \cdot A \cdot ds = p \cdot dV $ \par \par
	\textbf{Tekniskt arbete: } $ dW_t = - p  \cdot dV \text{; }  dw_t = - p  \cdot dv $\\		 
	Miunustecknet definierat att avgivet arbete vid expansion är \textgreater 0 
	\begin{flalign*}
    & \textbf{Entalpi: }h = u + p \cdot v &\\
    & dq = dh - v \cdot dp &
	\end{flalign*}
	\textbf{Entropi: } $ds = \left (\dfrac{dq}{dT}\right)_{rev}$ 
	\begin{flalign*}
   & \textbf{Gibbs fria energi: } G = H - T \cdot S \\
	 & \Delta G = \Delta H - T \cdot \Delta S \text{ för en isoterm process.}&
	\end{flalign*}
	\begin{flalign*}
   & \textbf{Exergi: } X_{\text{värme}}=\left( 1-\dfrac{T_{\text{omgivning}}}{T_{\text{källa}}}\right)\cdot Q & \\
   & X_{\text{arbete}}=W_t = W - p \cdot (V_2 - V_1) &
	\end{flalign*}
\section*{Termodynamiska processer}
	\begin{flalign*}
    &\textbf{Isokor process: } V = konstant & \\
		&Q = m \cdot c_v \cdot \Delta T & \\
		&w = 0 \\
		&w_t = v \cdot \Delta p... & 
	\end{flalign*}
	\begin{flalign*}
    &\textbf{Isobar process: } p = konstant & \\
		&Q = m \cdot c_p \cdot \Delta T & \\
		&w_t = 0 & 
	\end{flalign*}
	\begin{flalign*}
    & \textbf{Isoterm process: }  T = konstant & \\
		& Q = W = W_t = m \cdot R \cdot T \cdot \ln \dfrac{p_1}{p_2}= & \\
		& = m \cdot R \cdot T \cdot \ln \dfrac{v_2}{v_1} & \\
		&  w = R \cdot T \cdot \ln \dfrac{p_1}{p_2}=R \cdot T \cdot \ln \dfrac{v_2}{v_1}= T \cdot (s_2 - s_1) \text{ för en ideal gas.} & 
	\end{flalign*}  
	\begin{flalign*}  
		&  \textbf{Adiabat och ise}  \textbf {ntrop process: } & \\ 
		&  \Delta  Q = 0 \text{ och } \Delta s = 0 & \\
		& q=0 & \\
		&   W = \dfrac{p_1 \cdot V_1 - p_2 \cdot V_2}{\kappa - 1} = m \cdot \dfrac {R \cdot (T_1 - T_2)}{\kappa - 1} = U_1 - U_2\\
		& w_t = h_2 - h_1 = \kappa \cdot w
	\end{flalign*}
	\begin{flalign*}  
		&  \textbf{Polytrop process: } p\cdot V^n = \textit{kontant}& \\ 
		&   Q = m \cdot c_n \cdot (T_2 - T_1) \text{ där } c_n = -c_v \cdot \dfrac{{\kappa - n}}{n - 1}\\
		&   W = \dfrac{p_1 \cdot V_1 - p_2 \cdot V_2}{n - 1} = m \cdot \dfrac {R \cdot (T_1 - T_2)}{n - 1} \\
		&   W_t = n \cdot W
	\end{flalign*}
  \textbf{Carnotprocess: } $ \eta_c = \dfrac{\Delta s \cdot (T_1 - T_2)}{\Delta s \cdot (T_1 - 0)}= 1 - \dfrac{T_2}{T_1}$
