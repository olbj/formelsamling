\section*{Formler och samband.}
\subsection*{Allmänt}
\begin{flalign*}
	&\textit{Rotorns vinkelhastighet} &\\
	&\omega = \dfrac{2 \cdot \pi \cdot n}{60} \text{då n är varv per minut} \\
	&\textit{Löptal}\\
	&\lambda = \dfrac{\omega \cdot R}{v} = \dfrac{v_{spets}}{v} \text{där $v$ är den fria vindhastigheten}\\
	&\textit{Lokalt löptal}\\
	&\lambda = \dfrac{\omega \cdot r}{v} = \lambda \cdot \dfrac{r}{R} \text{där $r$ är avståndet till turbinaxeln}\\
	&\textit{Axeleffekt}\\
	& P = \omega \cdot M = P_{kin} \cdot C_p = \dfrac{\varrho}{2} \cdot A \cdot v^3 \cdot 4 \cdot a \cdot (1-a)^2 \\
	&\textit{Totalverkningsgrad} \\
	 &C_e = C_p \cdot \eta_{\text{växel}} \cdot \eta_{\text{generator}} \\
	&\textit{Eleffekt}\\
	&P_{el} = P_{kin} \cdot C_e 
\end{flalign*}
\subsection*{Vindens energiflöde}
\begin{flalign*}
	&\textit{Vindens momentana effekt} &\\
	&P_{kin} = \dfrac{varrho}{2} \cdot A \cdot v^3 \\
	&\textit{Vindens medeleffekt} \\
	&\overline{P_{kin}} = \dfrac{\varrho}{2} \cdot A \cdot \overline{v^3} \\
	&\textit{Vindens medeleffekt på en plats} \\
	&\overline{P_{kin}} = \dfrac{varrho}{2} \cdot A \cdot (\bar{v})^3 \cdot {EPF} \\
	&\textit{Kubikfaktor (Energy Pattern Factor)}\\
	&{EPF} = \dfrac{\overline{v^3}}{(\bar{v})^3} \\
	&\textit{Turbulensintensitet (10 minuters mätning, samplingsfrekvens minst 1 Hz)}\\
	&{TI} = \dfrac{{sd}(v)}{\bar{v}}
\end{flalign*}
\textbf{Vindhastighetens höjdberoende}
\begin{flalign*}
	&v_2 = v_1 \cdot \left(\dfrac{h_2 - h_0}{h_1 - h_0} \right)^\alpha &\\
	&h_0 = \text{ nollplansförskjutning } \approx \text{ 75\% av vegetationshöjden} \\
	&\alpha \approx 0,1 \text{ (hav och stora sjöar) till } 0,3 \text{ (höga träd och större städer).}
\end{flalign*}
\subsection*{Ideal turbin}
\begin{flalign*}
	&\textit{Axiella induktionsfaktorn} & \\
	& a = 1- \dfrac{v_{turbin}}{v}\\
	&\text{där } v_{turbin} = \text{ vindhastighet genom turbinen} \\ 
	&\text{och } v = \text{ den ostörda vindens hastighet}\\
	&\textit{Effektkoefficient}\\
	&C_p = \dfrac{P}{P_{kin}} = 4 \cdot a \cdot (1-a)^2 \\
	&\textit{Maximal effektkoefficient}\\
	&C_{p,max} = \dfrac{16}{27} \approx 0.5936 \ \text{när} \ a = \frac{1}{3}\\
	&\textit{Massflöde genom turbin}\\
	&\dot{m} = \varrho \cdot A \cdot v_{\text{turbin}} = \varrho \cdot A \cdot v \cdot (1-a) \\
	&\textit{Vältkraft} \\
	&F = \dfrac{\varrho}{2} \cdot A \cdot v^2 \cdot 4 \cdot a \cdot (1-a)
\end{flalign*}
