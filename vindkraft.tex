\section*{Formler och samband}
\subsection*{Vindens energiflöde}
\begin{flalign*}
	&\textit{Vindens momentana effekt} \\
	&P_{kin} = \dfrac{\rho}{2} \cdot A \cdot v^3 \\
	&\textit{Vindens medeleffekt} \\
	&\overline{P_{kin}} = \dfrac{\rho}{2} \cdot A \cdot \overline{v^3} \\
	&\textit{Vindens medeleffekt på en plats} \\
	&\overline{P_{kin}} = \dfrac{\rho}{2} \cdot A \cdot \bar{v}^3 \cdot {EPF} \\
	&\textit{Kubikfaktor (Energy Pattern Factor)}\\
	&{EPF} = \dfrac{\overline{v^3}}{\bar{v}^3} \\
	&\textit{Turbulensintensitet (10 minuters mätning, samplingsfrekvens minst 1 Hz)}\\
	&{TI} = \dfrac{{sd}(v)}{\bar{v}}
\end{flalign*}

\end{flalign*}
